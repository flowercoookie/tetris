// 테트리스 게임 로직 구현 const canvas = document.getElementById('tetrisCanvas'); const ctx = canvas.getContext('2d'); const ROWS = 20; const COLS = 10; const BLOCK_SIZE = canvas.height / ROWS; // 게임 상태 초기화 let board = []; for (let r = 0; r < ROWS; r++) { board[r] = []; for (let c = 0; c < COLS; c++) { board[r][c] = 0; } } // 블록 그리기 함수 function drawBlock(x, y, color) { ctx.fillStyle = color; ctx.fillRect(x * BLOCK_SIZE, y * BLOCK_SIZE, BLOCK_SIZE, BLOCK_SIZE); ctx.strokeStyle = '#333'; ctx.strokeRect(x * BLOCK_SIZE, y * BLOCK_SIZE, BLOCK_SIZE, BLOCK_SIZE); } // 테트리스 블록 클래스 class Tetromino { constructor(shape, color) { this.shape = shape; this.color = color; this.x = 4; // 초기 위치 설정 this.y = 0; } draw() { this.shape.forEach((row, r) => { row.forEach((val, c) => { if (val) { drawBlock(this.x + c, this.y + r, this.color); } }); }); } } // 테트리스 게임 메인 함수 function main() { // 게임 로직 추가 } main(); // 게임 시작
